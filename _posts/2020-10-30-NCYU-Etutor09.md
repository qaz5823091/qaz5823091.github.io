```c++
#include <iostream>  
using namespace std;  
int main()  
{  
    int n;  
    cin>>n;  
    int center = n/2+1;  
    int offset = 0;  
    for(int row=1;row<=n;row++){  
        for(int col=1;col<=n;col++){  
            if ( col>center+offset || col<center-offset )  
                cout<<"x";  
            else  
                cout<<"o";  
        }  
        if ( row<center )  
            offset++;  
        else  
            offset--;  

        cout<<endl;  
    }  
}
```
