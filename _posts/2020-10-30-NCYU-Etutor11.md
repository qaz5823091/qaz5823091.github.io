```c++
#include <iostream>  
using namespace std;  
int main()  
{  
    while(1){  
        int n;  
        cin>>n;  
        if ( n == -1 )  
            break;  

        int counter = 0;  
        while(1){  
            if ( n%2==0 ){  
                counter++;  
                n /= 2;  
            }  
            else  
                break;  
        }  
        cout<<counter<<endl;  
    }  
}
```
