```c++
#include <iostream>  
#include <algorithm>  
using namespace std;  
int main()  
{  
    int t;  
    cin>>t;  
    while(t>0){  
        int number[3];  
        cin>>number[0]>>number[1]>>number[2];  
        sort(number,number+3);  
        reverse(number,number+3);  
        cout<<number[0]<<" "<<number[1]<<" "<<number[2]<<endl;  

        t--;  
    }  
}
```
