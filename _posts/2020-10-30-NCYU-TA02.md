```c++
#include <iostream>  
using namespace std;  
int main()  
{  
    string number;  
    cin>>number;  
    int length = number.size();  
    for(int i=length-1;i>=0;i--){  
        if ( number[i]-48!=0 ){  
            length = i;  
            break;  
        }  
    }  
    for(int i=length;i>=0;i--)  
        cout<<number[i];  
    cout<<endl;  
}
```
