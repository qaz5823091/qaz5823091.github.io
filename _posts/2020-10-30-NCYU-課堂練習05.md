```c++
#include <iostream>
using namespace std;
int main()
{
    int x,y;
    cin>>x>>y;

    if ( x!=0 && y!=0 ){
        if ( x>0 ){
            if ( y>0 )
                cout<<"第一象限";
            else
                cout<<"第四象限";
        }
        else{
            if ( y>0 )
                cout<<"第二象限";
            else
                cout<<"第三象限";
        }
    }
    else{
        if ( x==0 && y==0 )
            cout<<"原點";
        else if ( x==0 )
            cout<<"Y軸";
        else
            cout<<"X軸";
    }
}
```

```c++
#include <iostream>
using namespace std;
int main()
{
    char word;
    cin>>word;

    // a 97 z 122 A 65 Z 90
    if ( 97<=word && word<=122 )
        cout<<"Character "<<word<<" is a lowercase letter.";
    else
        cout<<"Character "<<word<<" is a uppercase letter.";
}
```

```c++
#include <iostream>
using namespace std;
int main()
{
    string word;
    getline(cin,word);

    // a 97 z 122 A 65 Z 90
    for(unsigned int i=0;i<word.size();i++){
        if ( 97<=word.at(i) && word.at(i)<=122 )
            word.at(i) -= 32;
    }
    cout<<word;
}
```
