```c++
#include <iostream>  
using namespace std;  

int gcd(int x, int y){  
    while(y!=0){  
        int tmp = x%y;  
        x = y;  
        y = tmp;  
    }  
    return x;  
};  


int main()  
{  
    int a,b;  
    while(cin>>a>>b){  
        int GCD = gcd(a,b);  
        cout<<GCD<<" "<<a*b/GCD<<endl;  
    }  
}
```
